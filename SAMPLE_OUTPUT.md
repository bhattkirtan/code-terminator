import { Injectable } from '@angular/core';
import { BehaviorSubject, Observable, of } from 'rxjs';
import { delay, map } from 'rxjs/operators';

export interface CaseItem {
  id: string;
  title: string;
  description: string;
  status: 'active' | 'pending' | 'completed' | 'cancelled';
  createdAt: Date;
  updatedAt: Date;
}

@Injectable({
  providedIn: 'root'
})
export class CasesPortalService {
  private dataSubject = new BehaviorSubject<CaseItem[]>([]);
  private loadingSubject = new BehaviorSubject<boolean>(false);
  
  public data$ = this.dataSubject.asObservable();
  public loading$ = this.loadingSubject.asObservable();

  constructor() {
    this.loadInitialData();
  }

  /**
   * Load initial mock data
   */
  private loadInitialData(): void {
    const mockData: CaseItem[] = [
      {
        id: '1',
        title: 'Project Scope Definition',
        description: 'Define the scope and parameters for the new project',
        status: 'active',
        createdAt: new Date(),
        updatedAt: new Date()
      },
      {
        id: '2', 
        title: 'History Analysis Review',
        description: 'Analyze historical data and identify trends',
        status: 'pending',
        createdAt: new Date(),
        updatedAt: new Date()
      }
    ];
    
    this.dataSubject.next(mockData);
  }

  /**
   * Get all cases
   */
  public getCases(): Observable<CaseItem[]> {
    this.loadingSubject.next(true);
    
    return of(this.dataSubject.value).pipe(
      delay(500), // Simulate network delay
      map(items => {
        this.loadingSubject.next(false);
        return items;
      })
    );
  }

  /**
   * Create new case
   */
  public createCase(item: Partial<CaseItem>): Observable<CaseItem> {
    const newItem: CaseItem = {
      id: Date.now().toString(),
      title: item.title || 'New Case',
      description: item.description || '',
      status: item.status || 'pending',
      createdAt: new Date(),
      updatedAt: new Date()
    };

    const currentData = this.dataSubject.value;
    this.dataSubject.next([...currentData, newItem]);

    return of(newItem).pipe(delay(300));
  }
}
